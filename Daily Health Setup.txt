git clone https://github.com/kong125712/Daily-health.git
cd ./Daily-health

npm install

# --- 创建 .env.local（项目根目录，和 package.json 同级）---
# 内容示例：
# DATABASE_URL="file:./daily-health.db"
# AI_PROVIDER=gemini
# OPENAI_API_KEY=
# GEMINI_API_KEY=你的key
# EPICURE_MCP_URL=https://epicure-mcp.kaikaku.ai/mcp
#
# 用终端命令写，不要用图形界面编辑器（避免RTF格式污染文件）：
cat > .env.local << 'EOF'
DATABASE_URL="file:./daily-health.db"
AI_PROVIDER=gemini
OPENAI_API_KEY=
GEMINI_API_KEY=
EPICURE_MCP_URL=https://epicure-mcp.kaikaku.ai/mcp
EOF

# 写完务必检查内容是否干净（没有反斜杠、引号、隐藏字符）：
cat .env.local

# Prisma CLI 不会自动读 .env.local，只认 .env，建软链接让两边都能读到同一份配置：
ln -s .env.local .env

npm uninstall prisma @prisma/client
npm uninstall @prisma/adapter-libsql @libsql/client
npm install prisma@5.22.0
npm install @prisma/client@5.22.0

npx prisma generate --schema database/schema.prisma
npx prisma migrate dev --schema database/schema.prisma

npm run dev
